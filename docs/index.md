max = progressbar.attr('max'), 
time = (300/max)*5,

value += 0.5; 
addValue = progressbar.val(value); 

$('.progress-value').html(value + '%'); 

if (value == max) {

} 
}; 

var animate = setInterval(function() { 
loading(); 
}, time); 
}); 

</script>
